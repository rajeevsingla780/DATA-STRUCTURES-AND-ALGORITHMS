//Array Subset of another array

//-------------------------------------------------
//Code in cpp O(n)

string isSubset(int a1[], int a2[], int n, int m) {
    unordered_map<int,int> m1;
    bool t=true;
    for(int i=0;i<n;i++)
     m1[a1[i]]++;
     
     for(int i=0;i<m;i++)
     {
         if(m1.find(a2[i])==m1.end())
         return "No";
         else if(m1.find(a2[i])!=m1.end()&&m1[a2[i]]!=0)
             m1[a2[i]]--;
     }
     return "Yes";
    
}
